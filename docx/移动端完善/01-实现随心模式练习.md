### 步骤1：创建练习页面文件

**文件路径：** `apps/web/src/features/mobile/pages/freestyle/MobileFreestylePracticePage.tsx`

**完整代码：**（请完全复制，不要修改）

```typescript
import { useState, useEffect, useCallback } from 'react'
import { useNavigate } from 'react-router-dom'
import { X, ChevronRight, RotateCcw } from 'lucide-react'
import { MobileLayout, MobileTouchButton } from '@/features/mobile/components/layout/MobileLayout'
import {
  readFreestyleConfig,
  readFreestyleProgress,
  buildFreestyleQueue,
  saveFreestyleProgress,
  type FreestyleConfig,
  type FreestyleProgressSnapshot,
} from '@/features/freestyle/model/freestyle'

interface FreestyleCard {
  id: number
  question: string
  answer: string
  type: 'qa' | 'mindmap'
}

/**
 * 移动端随心模式练习页面
 * 全屏显示卡片，支持翻转查看答案
 */
export default function MobileFreestylePracticePage() {
  const navigate = useNavigate()
  const [config, setConfig] = useState<FreestyleConfig | null>(null)
  const [progress, setProgress] = useState<FreestyleProgressSnapshot | null>(null)
  const [queue, setQueue] = useState<FreestyleCard[]>([])
  const [currentIndex, setCurrentIndex] = useState(0)
  const [isFlipped, setIsFlipped] = useState(false)
  const [isLoading, setIsLoading] = useState(true)

  // 加载配置和构建队列
  useEffect(() => {
    const loadData = async () => {
      try {
        setIsLoading(true)
        const [loadedConfig, loadedProgress] = await Promise.all([
          readFreestyleConfig(),
          readFreestyleProgress(),
        ])
        
        setConfig(loadedConfig)
        setProgress(loadedProgress)

        // 构建练习队列
        const builtQueue = buildFreestyleQueue(loadedConfig, {
          resolvedQuestionIds: loadedProgress.resolvedQuestionIds,
        })

        // 转换为简化的卡片格式
        const cards: FreestyleCard[] = builtQueue.questions.map((q, index) => ({
          id: q.id || index,
          question: q.prompt || '问题加载失败',
          answer: q.correctAnswer || '答案加载失败',
          type: 'qa',
        }))

        setQueue(cards)
        setCurrentIndex(loadedProgress.currentIndex || 0)
      } catch (error) {
        console.error('Failed to load freestyle practice:', error)
      } finally {
        setIsLoading(false)
      }
    }

    void loadData()
  }, [])

  // 翻转卡片
  const handleFlip = useCallback(() => {
    setIsFlipped(!isFlipped)
  }, [isFlipped])

  // 下一张卡片
  const handleNext = useCallback(async () => {
    if (!progress || !config) return

    const nextIndex = currentIndex + 1

    // 保存进度
    const updatedProgress = {
      ...progress,
      currentIndex: nextIndex,
      resolvedQuestionIds: [
        ...progress.resolvedQuestionIds,
        queue[currentIndex]?.id || 0,
      ],
    }

    await saveFreestyleProgress(updatedProgress)
    setProgress(updatedProgress)

    // 如果是最后一张，返回主页
    if (nextIndex >= queue.length) {
      navigate('/m/freestyle')
      return
    }

    // 切换到下一张
    setCurrentIndex(nextIndex)
    setIsFlipped(false)
  }, [currentIndex, queue, progress, config, navigate])

  // 退出练习
  const handleExit = useCallback(() => {
    if (confirm('确定要退出练习吗？进度已自动保存。')) {
      navigate('/m/freestyle')
    }
  }, [navigate])

  // 重新翻到正面
  const handleReset = useCallback(() => {
    setIsFlipped(false)
  }, [])

  if (isLoading) {
    return (
      <MobileLayout showTabBar={false}>
        <div className="flex items-center justify-center h-full">
          <div className="text-center">
            <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-primary mx-auto mb-4" />
            <p className="text-muted-foreground">加载中...</p>
          </div>
        </div>
      </MobileLayout>
    )
  }

  if (queue.length === 0) {
    return (
      <MobileLayout showTabBar={false}>
        <div className="flex flex-col items-center justify-center h-full p-6">
          <p className="text-lg text-muted-foreground mb-6">暂无练习内容</p>
          <MobileTouchButton variant="secondary" onClick={() => navigate('/m/freestyle')}>
            返回
          </MobileTouchButton>
        </div>
      </MobileLayout>
    )
  }

  const currentCard = queue[currentIndex]
  const progressPercent = ((currentIndex + 1) / queue.length) * 100

  return (
    <MobileLayout showTabBar={false} disableScroll>
      <div className="fixed inset-0 flex flex-col bg-background">
        {/* 顶部栏 */}
        <div className="flex items-center justify-between px-4 py-3 border-b border-border">
          <button
            type="button"
            onClick={handleExit}
            className="p-2 rounded-full active:bg-secondary"
            aria-label="退出"
          >
            <X className="w-6 h-6" />
          </button>

          <div className="flex-1 mx-4">
            <div className="text-sm text-muted-foreground text-center mb-1">
              {currentIndex + 1} / {queue.length}
            </div>
            <div className="w-full bg-secondary rounded-full h-2">
              <div
                className="bg-primary rounded-full h-2 transition-all duration-300"
                style={{ width: `${progressPercent}%` }}
              />
            </div>
          </div>

          <button
            type="button"
            onClick={handleReset}
            className="p-2 rounded-full active:bg-secondary"
            aria-label="重置"
          >
            <RotateCcw className="w-5 h-5" />
          </button>
        </div>

        {/* 卡片区域 */}
        <div className="flex-1 flex items-center justify-center p-6">
          <div
            onClick={handleFlip}
            className="w-full max-w-md aspect-[3/4] bg-card border-2 border-border rounded-3xl shadow-lg flex flex-col items-center justify-center p-8 cursor-pointer active:scale-95 transition-transform"
          >
            {!isFlipped ? (
              // 问题面
              <>
                <p className="text-xs text-muted-foreground uppercase tracking-wider mb-4">
                  问题
                </p>
                <p className="text-xl font-medium text-center leading-relaxed">
                  {currentCard.question}
                </p>
                <p className="text-sm text-muted-foreground mt-8">点击查看答案</p>
              </>
            ) : (
              // 答案面
              <>
                <p className="text-sm text-primary font-medium mb-4">答案</p>
                <p className="text-lg text-center leading-relaxed">
                  {currentCard.answer}
                </p>
              </>
            )}
          </div>
        </div>

        {/* 底部操作区 */}
        <div className="p-6 pb-8 space-y-3 border-t border-border">
          {isFlipped && (
            <MobileTouchButton
              variant="primary"
              onClick={handleNext}
              className="w-full flex items-center justify-center gap-2"
            >
              <span className="text-lg font-semibold">
                {currentIndex + 1 < queue.length ? '下一张' : '完成练习'}
              </span>
              <ChevronRight className="w-5 h-5" />
            </MobileTouchButton>
          )}
        </div>
      </div>
    </MobileLayout>
  )
}
```
