import { type ReactNode } from 'react'

interface MobileLayoutProps {
  /**
   * 页面主体内容
   */
  children: ReactNode

  /**
   * 顶部标题栏内容（可选）
   */
  topBar?: ReactNode

  /**
   * 是否显示底部 Tab Bar（默认 true）
   */
  showTabBar?: boolean

  /**
   * 自定义内容区域的 className
   */
  contentClassName?: string

  /**
   * 是否禁用内容区域滚动（默认 false）
   */
  disableScroll?: boolean
}

/**
 * 移动端通用布局组件
 * 
 * 功能：
 * - 处理 iOS safe area insets（刘海、Home 指示条）
 * - 为底部 Tab Bar 预留空间
 * - 提供统一的滚动容器
 * - 支持可选的顶部标题栏
 * 
 * 使用示例：
 * ```tsx
 * <MobileLayout topBar={<MobileTopBar title="随心模式" />}>
 *   <div>页面内容</div>
 * </MobileLayout>
 * ```
 */
export function MobileLayout({
  children,
  topBar,
  showTabBar = true,
  contentClassName = '',
  disableScroll = false,
}: MobileLayoutProps) {
  return (
    <div className="mobile-page-container">
      {/* 顶部标题栏（可选） */}
      {topBar && <div className="mobile-top-bar">{topBar}</div>}

      {/* 主内容区域 */}
      <div
        className={`mobile-content-area ${contentClassName} ${disableScroll ? 'overflow-hidden' : ''}`}
      >
        {children}
      </div>

      {/* 底部 Tab Bar 占位（实际 Tab Bar 由 MobileApp 渲染） */}
      {showTabBar && <div className="h-16" aria-hidden="true" />}
    </div>
  )
}

/**
 * 移动端顶部标题栏组件
 */
interface MobileTopBarProps {
  /**
   * 标题文本
   */
  title: string

  /**
   * 左侧按钮（可选）
   */
  leftAction?: ReactNode

  /**
   * 右侧按钮（可选）
   */
  rightAction?: ReactNode
}

export function MobileTopBar({ title, leftAction, rightAction }: MobileTopBarProps) {
  return (
    <>
      {/* 左侧按钮区域 */}
      <div className="flex items-center gap-2">
        {leftAction || <div className="w-10" />} {/* 占位，保持标题居中 */}
      </div>

      {/* 标题 */}
      <h1 className="text-lg font-semibold text-foreground truncate">{title}</h1>

      {/* 右侧按钮区域 */}
      <div className="flex items-center gap-2">
        {rightAction || <div className="w-10" />} {/* 占位，保持标题居中 */}
      </div>
    </>
  )
}

/**
 * 移动端触控友好按钮
 */
interface MobileTouchButtonProps {
  /**
   * 按钮内容
   */
  children: ReactNode

  /**
   * 点击事件
   */
  onClick?: () => void

  /**
   * 按钮类型
   */
  variant?: 'primary' | 'secondary' | 'ghost'

  /**
   * 是否禁用
   */
  disabled?: boolean

  /**
   * 自定义 className
   */
  className?: string

  /**
   * 按钮类型（HTML button type）
   */
  type?: 'button' | 'submit' | 'reset'
}

export function MobileTouchButton({
  children,
  onClick,
  variant = 'primary',
  disabled = false,
  className = '',
  type = 'button',
}: MobileTouchButtonProps) {
  const baseClasses = 'mobile-touch-button'
  
  const variantClasses = {
    primary: 'bg-primary text-primary-foreground',
    secondary: 'bg-secondary text-secondary-foreground',
    ghost: 'bg-transparent text-foreground',
  }

  return (
    <button
      type={type}
      onClick={onClick}
      disabled={disabled}
      className={`${baseClasses} ${variantClasses[variant]} ${className} ${disabled ? 'opacity-50 cursor-not-allowed' : ''}`}
    >
      {children}
    </button>
  )
}

/**
 * 移动端卡片组件
 */
interface MobileCardProps {
  /**
   * 卡片内容
   */
  children: ReactNode

  /**
   * 点击事件（可选，如果提供则卡片可点击）
   */
  onClick?: () => void

  /**
   * 自定义 className
   */
  className?: string
}

export function MobileCard({ children, onClick, className = '' }: MobileCardProps) {
  const Component = onClick ? 'button' : 'div'
  
  return (
    <Component
      onClick={onClick}
      className={`mobile-card ${onClick ? 'cursor-pointer' : ''} ${className}`}
    >
      {children}
    </Component>
  )
}

/**
 * 移动端列表项组件
 */
interface MobileListItemProps {
  /**
   * 列表项内容
   */
  children: ReactNode

  /**
   * 点击事件（可选）
   */
  onClick?: () => void

  /**
   * 左侧图标（可选）
   */
  icon?: ReactNode

  /**
   * 右侧箭头（可选）
   */
  showArrow?: boolean

  /**
   * 自定义 className
   */
  className?: string
}

export function MobileListItem({
  children,
  onClick,
  icon,
  showArrow = false,
  className = '',
}: MobileListItemProps) {
  const Component = onClick ? 'button' : 'div'
  
  return (
    <Component
      onClick={onClick}
      className={`mobile-list-item ${onClick ? 'cursor-pointer' : ''} ${className}`}
    >
      {/* 左侧图标 */}
      {icon && <div className="flex-shrink-0">{icon}</div>}

      {/* 主内容 */}
      <div className="flex-1 min-w-0">{children}</div>

      {/* 右侧箭头 */}
      {showArrow && (
        <svg
          className="w-5 h-5 text-muted-foreground flex-shrink-0"
          fill="none"
          stroke="currentColor"
          viewBox="0 0 24 24"
        >
          <path
            strokeLinecap="round"
            strokeLinejoin="round"
            strokeWidth={2}
            d="M9 5l7 7-7 7"
          />
        </svg>
      )}
    </Component>
  )
}

/**
 * 移动端空状态组件
 */
interface MobileEmptyStateProps {
  /**
   * 图标
   */
  icon?: ReactNode

  /**
   * 标题
   */
  title: string

  /**
   * 描述文本（可选）
   */
  description?: string

  /**
   * 操作按钮（可选）
   */
  action?: ReactNode
}

export function MobileEmptyState({
  icon,
  title,
  description,
  action,
}: MobileEmptyStateProps) {
  return (
    <div className="mobile-empty-state">
      {icon}
      <div className="text-base font-medium text-foreground">{title}</div>
      {description && <div className="text-sm text-muted-foreground">{description}</div>}
      {action && <div className="mt-4">{action}</div>}
    </div>
  )
}

## 组件说明

### MobileLayout
主布局组件，所有移动端页面都应该包裹在这个组件中。

**Props:**
- `children` - 页面主体内容
- `topBar` - 可选的顶部标题栏
- `showTabBar` - 是否为底部 Tab Bar 预留空间（默认 true）
- `contentClassName` - 自定义内容区域样式
- `disableScroll` - 禁用滚动（某些特殊页面需要）

### MobileTopBar
标准的移动端顶部标题栏。

**Props:**
- `title` - 标题文本
- `leftAction` - 左侧按钮（如返回按钮）
- `rightAction` - 右侧按钮（如设置按钮）

### MobileTouchButton
触控友好的按钮，满足 44px 最小触控尺寸。

**Props:**
- `variant` - 样式变体：primary / secondary / ghost
- `disabled` - 是否禁用
- `onClick` - 点击回调

### MobileCard
移动端卡片组件，支持点击交互。

### MobileListItem
列表项组件，支持图标、箭头、点击。

### MobileEmptyState
空状态展示组件。
