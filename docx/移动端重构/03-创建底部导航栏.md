```typescript
import { Home, BookOpen, RotateCcw, Globe, User } from 'lucide-react'
import { useLocation, useNavigate } from 'react-router-dom'

/**
 * Tab 配置项
 */
interface TabConfig {
  /**
   * Tab ID（唯一标识）
   */
  id: string

  /**
   * Tab 显示名称
   */
  label: string

  /**
   * Tab 图标组件
   */
  icon: typeof Home

  /**
   * Tab 对应的路由路径
   */
  path: string

  /**
   * 路径匹配模式（用于判断当前激活状态）
   */
  pathPattern: RegExp
}

/**
 * 移动端底部 Tab 配置
 */
const MOBILE_TABS: TabConfig[] = [
  {
    id: 'freestyle',
    label: '随心',
    icon: Home,
    path: '/m/freestyle',
    pathPattern: /^\/m(\/freestyle)?$/,
  },
  {
    id: 'palaces',
    label: '宫殿',
    icon: BookOpen,
    path: '/m/palaces',
    pathPattern: /^\/m\/palaces/,
  },
  {
    id: 'review',
    label: '复习',
    icon: RotateCcw,
    path: '/m/review',
    pathPattern: /^\/m\/review/,
  },
  {
    id: 'english',
    label: '英语',
    icon: Globe,
    path: '/m/english',
    pathPattern: /^\/m\/english/,
  },
  {
    id: 'profile',
    label: '我的',
    icon: User,
    path: '/m/profile',
    pathPattern: /^\/m\/profile/,
  },
]

/**
 * 移动端底部 Tab Bar 导航栏
 * 
 * 功能：
 * - 5个固定 Tab：随心、宫殿、复习、英语、我的
 * - 自动高亮当前激活的 Tab
 * - 支持触感反馈（Haptic Feedback）
 * - 响应式布局，适配不同屏幕宽度
 * - 处理 iOS safe area（底部 Home 指示条）
 * 
 * 使用：
 * 在 MobileApp.tsx 中直接使用，无需传参
 */
export function MobileTabBar() {
  const location = useLocation()
  const navigate = useNavigate()

  /**
   * 判断某个 Tab 是否为当前激活状态
   */
  const isTabActive = (tab: TabConfig): boolean => {
    return tab.pathPattern.test(location.pathname)
  }

  /**
   * 处理 Tab 点击事件
   */
  const handleTabClick = (tab: TabConfig) => {
    // 如果点击的是当前激活的 Tab，滚动到顶部
    if (isTabActive(tab)) {
      window.scrollTo({ top: 0, behavior: 'smooth' })
      return
    }

    // 触发触感反馈（iOS）
    try {
      if ('vibrate' in navigator) {
        navigator.vibrate(10)
      }
    } catch (error) {
      // 忽略触感反馈错误
    }

    // 导航到目标页面
    navigate(tab.path)
  }

  return (
    <nav className="mobile-tab-bar" role="navigation" aria-label="主导航">
      <div className="mobile-tab-bar-inner">
        {MOBILE_TABS.map((tab) => {
          const Icon = tab.icon
          const isActive = isTabActive(tab)

          return (
            <button
              key={tab.id}
              type="button"
              onClick={() => handleTabClick(tab)}
              className={`mobile-tab-button ${isActive ? 'active' : ''}`}
              aria-label={tab.label}
              aria-current={isActive ? 'page' : undefined}
            >
              {/* Tab 图标 */}
              <Icon strokeWidth={isActive ? 2.5 : 2} />

              {/* Tab 文字 */}
              <span>{tab.label}</span>
            </button>
          )
        })}
      </div>
    </nav>
  )
}

/**
 * 判断当前路径是否应该显示 Tab Bar
 * 
 * 某些特殊页面（如全屏播放、编辑器）可能不需要显示 Tab Bar
 * 
 * @param pathname - 当前路由路径
 * @returns 是否显示 Tab Bar
 */
export function shouldShowTabBar(pathname: string): boolean {
  // 全屏模式路径（不显示 Tab Bar）
  const fullscreenPaths = [
    /^\/m\/palaces\/\d+\/practice/, // 宫殿练习时全屏
    /^\/m\/review\/session/, // 复习会话时全屏
  ]

  return !fullscreenPaths.some((pattern) => pattern.test(pathname))
}
```

## 组件说明

### MobileTabBar
固定在底部的导航栏，包含5个 Tab。

**功能特性：**
1. **自动高亮**：根据当前路由自动高亮对应的 Tab
2. **触感反馈**：点击 Tab 时触发轻微振动（iOS）
3. **滚动到顶**：点击当前激活的 Tab 会平滑滚动到页面顶部
4. **无障碍支持**：正确使用 ARIA 属性

**Tab 列表：**
| Tab ID | 名称 | 图标 | 路径 | 匹配规则 |
|--------|------|------|------|----------|
| freestyle | 随心 | Home | /m/freestyle | /m 或 /m/freestyle |
| palaces | 宫殿 | BookOpen | /m/palaces | /m/palaces* |
| review | 复习 | RotateCcw | /m/review | /m/review* |
| english | 英语 | Globe | /m/english | /m/english* |
| profile | 我的 | User | /m/profile | /m/profile* |

### shouldShowTabBar
工具函数，判断当前页面是否应该显示 Tab Bar。

**隐藏 Tab Bar 的场景：**
- 宫殿练习时（全屏模式）
- 复习会话时（全屏模式）

## 使用示例

```tsx
// 在 MobileApp.tsx 中使用
import { MobileTabBar, shouldShowTabBar } from './components/nav/MobileTabBar'
import { useLocation } from 'react-router-dom'

function MobileApp() {
  const location = useLocation()
  const showTabBar = shouldShowTabBar(location.pathname)

  return (
    <div>
      {/* 页面路由 */}
      <Routes>
        {/* ... */}
      </Routes>

      {/* 底部 Tab Bar */}
      {showTabBar && <MobileTabBar />}
    </div>
  )
}
```

## 交互行为

### 点击未激活的 Tab
1. 触发轻微振动（iOS，10ms）
2. 导航到对应路由
3. Tab 按钮缩放动画（scale: 0.95）

### 点击已激活的 Tab
1. 不触发导航
2. 页面平滑滚动到顶部
3. 无振动反馈

### 视觉反馈
- **激活状态**：图标和文字变为主题色（primary）
- **未激活**：灰色（muted-foreground）
- **点击时**：按钮轻微缩放（active state）
