使用 Write 工具创建 `apps/web/src/features/mobile/pages/freestyle/MobileFreestylePage.tsx`：

```typescript
import { useState, useEffect, useCallback } from 'react'
import { Play, Settings, Sparkles, RotateCcw } from 'lucide-react'
import {
  MobileLayout,
  MobileTopBar,
  MobileTouchButton,
  MobileCard,
  MobileEmptyState,
} from '@/features/mobile/components/layout/MobileLayout'
import {
  readFreestyleConfig,
  readFreestyleProgress,
  buildFreestyleQueue,
  nextFreestyleSeed,
  saveFreestyleProgress,
  type FreestyleConfig,
  type FreestyleProgressSnapshot,
} from '@/features/freestyle/model/freestyle'

/**
 * 移动端随心模式页面
 * 
 * 核心功能：
 * - 随机卡片练习（问答、脑图）
 * - 今日训练模式
 * - 练习进度追踪
 * - 简化的配置面板
 * 
 * 移动端优化：
 * - 大号触控按钮
 * - 全屏卡片展示
 * - 手势支持（滑动切换卡片）
 * - 简化的设置面板
 */
export default function MobileFreestylePage() {
  const [config, setConfig] = useState<FreestyleConfig | null>(null)
  const [progress, setProgress] = useState<FreestyleProgressSnapshot | null>(null)
  const [isLoading, setIsLoading] = useState(true)
  const [currentCardIndex, setCurrentCardIndex] = useState(0)
  const [showSettings, setShowSettings] = useState(false)

  // 初始化：加载配置和进度
  useEffect(() => {
    const loadData = async () => {
      try {
        setIsLoading(true)
        const [loadedConfig, loadedProgress] = await Promise.all([
          readFreestyleConfig(),
          readFreestyleProgress(),
        ])
        setConfig(loadedConfig)
        setProgress(loadedProgress)
      } catch (error) {
        console.error('Failed to load freestyle data:', error)
      } finally {
        setIsLoading(false)
      }
    }

    void loadData()
  }, [])

  // 开始练习
  const handleStartPractice = useCallback(() => {
    if (!config) return

    // TODO: 实现练习逻辑
    // 1. 构建练习队列
    // 2. 导航到卡片展示页面
    console.log('Start practice with config:', config)
  }, [config])

  // 打开设置
  const handleOpenSettings = useCallback(() => {
    setShowSettings(true)
  }, [])

  // 重置进度
  const handleResetProgress = useCallback(async () => {
    if (!confirm('确定要重置今日进度吗？')) return

    try {
      const newSeed = nextFreestyleSeed()
      await saveFreestyleProgress({
        seed: newSeed,
        completedIndices: [],
        skippedIndices: [],
      })
      setProgress({
        seed: newSeed,
        completedIndices: [],
        skippedIndices: [],
      })
    } catch (error) {
      console.error('Failed to reset progress:', error)
    }
  }, [])

  // 加载状态
  if (isLoading) {
    return (
      <MobileLayout>
        <div className="flex items-center justify-center h-full">
          <div className="text-center">
            <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-primary mx-auto mb-4" />
            <p className="text-muted-foreground">加载中...</p>
          </div>
        </div>
      </MobileLayout>
    )
  }

  // 配置未加载
  if (!config) {
    return (
      <MobileLayout>
        <MobileEmptyState
          icon={<Sparkles className="w-16 h-16" />}
          title="随心模式未配置"
          description="请先在桌面端完成初始配置"
        />
      </MobileLayout>
    )
  }

  const completedCount = progress?.completedIndices.length || 0
  const totalCount = 20 // TODO: 从配置计算总数

  return (
    <MobileLayout
      topBar={
        <MobileTopBar
          title="随心模式"
          rightAction={
            <button
              type="button"
              onClick={handleOpenSettings}
              className="p-2 rounded-full active:bg-secondary"
              aria-label="设置"
            >
              <Settings className="w-5 h-5" />
            </button>
          }
        />
      }
    >
      <div className="flex flex-col gap-4 mobile-px mobile-py">
        {/* 进度卡片 */}
        <MobileCard>
          <div className="flex items-center justify-between mb-4">
            <div>
              <h2 className="text-lg font-semibold">今日进度</h2>
              <p className="text-sm text-muted-foreground">
                已完成 {completedCount} / {totalCount}
              </p>
            </div>
            <button
              type="button"
              onClick={handleResetProgress}
              className="p-2 rounded-full active:bg-secondary"
              aria-label="重置进度"
            >
              <RotateCcw className="w-5 h-5 text-muted-foreground" />
            </button>
          </div>

          {/* 进度条 */}
          <div className="w-full bg-secondary rounded-full h-3">
            <div
              className="bg-primary rounded-full h-3 transition-all duration-300"
              style={{ width: `${(completedCount / totalCount) * 100}%` }}
            />
          </div>
        </MobileCard>

        {/* 开始练习按钮 */}
        <MobileTouchButton
          variant="primary"
          onClick={handleStartPractice}
          className="w-full flex items-center justify-center gap-2"
        >
          <Play className="w-5 h-5" />
          <span className="text-lg font-semibold">开始练习</span>
        </MobileTouchButton>

        {/* 今日统计 */}
        <div className="grid grid-cols-2 gap-3">
          <MobileCard className="text-center">
            <div className="text-2xl font-bold text-primary">{completedCount}</div>
            <div className="text-sm text-muted-foreground mt-1">已完成</div>
          </MobileCard>

          <MobileCard className="text-center">
            <div className="text-2xl font-bold text-muted-foreground">
              {totalCount - completedCount}
            </div>
            <div className="text-sm text-muted-foreground mt-1">待练习</div>
          </MobileCard>
        </div>

        {/* 功能说明 */}
        <MobileCard>
          <h3 className="text-base font-semibold mb-3">关于随心模式</h3>
          <ul className="space-y-2 text-sm text-muted-foreground">
            <li>• 每日自动生成个性化练习</li>
            <li>• 混合宫殿记忆、英语、知识库内容</li>
            <li>• 支持 AI 智能解释</li>
            <li>• 记录学习轨迹和时长</li>
          </ul>
        </MobileCard>
      </div>

      {/* 设置面板（后续实现） */}
      {showSettings && (
        <div className="fixed inset-0 bg-black/50 z-50 flex items-end">
          <div className="bg-background w-full rounded-t-3xl p-6">
            <h2 className="text-lg font-semibold mb-4">随心模式设置</h2>
            <p className="text-sm text-muted-foreground mb-6">
              详细设置请在桌面端完成
            </p>
            <MobileTouchButton
              variant="secondary"
              onClick={() => setShowSettings(false)}
              className="w-full"
            >
              关闭
            </MobileTouchButton>
          </div>
        </div>
      )}
    </MobileLayout>
  )
}
```

### 步骤3：创建练习卡片组件（后续扩展）
使用 Write 工具创建 `apps/web/src/features/mobile/pages/freestyle/MobileFreestyleCard.tsx`：

```typescript
import { useState } from 'react'
import { ChevronRight, Lightbulb } from 'lucide-react'
import { MobileTouchButton } from '@/features/mobile/components/layout/MobileLayout'

interface MobileFreestyleCardProps {
  /**
   * 卡片类型
   */
  type: 'qa' | 'mindmap'

  /**
   * 问题文本
   */
  question: string

  /**
   * 答案文本
   */
  answer: string

  /**
   * 下一张卡片回调
   */
  onNext: () => void

  /**
   * 跳过卡片回调
   */
  onSkip: () => void

  /**
   * 请求 AI 解释回调
   */
  onAiExplain?: () => void
}

/**
 * 移动端随心模式卡片
 * 
 * 支持：
 * - 问答卡片
 * - 脑图卡片
 * - 翻转动画
 * - 滑动手势
 */
export function MobileFreestyleCard({
  type,
  question,
  answer,
  onNext,
  onSkip,
  onAiExplain,
}: MobileFreestyleCardProps) {
  const [isFlipped, setIsFlipped] = useState(false)

  const handleFlip = () => {
    setIsFlipped(!isFlipped)
  }

  return (
    <div className="flex flex-col h-full">
      {/* 卡片主体 */}
      <div className="flex-1 mobile-px mobile-py">
        <div
          onClick={handleFlip}
          className="mobile-card h-full flex flex-col justify-center items-center p-8 cursor-pointer"
        >
          {!isFlipped ? (
            // 问题面
            <div className="text-center">
              <p className="text-xs text-muted-foreground uppercase tracking-wider mb-4">
                {type === 'qa' ? '问答' : '脑图'}
              </p>
              <p className="text-xl font-medium leading-relaxed">{question}</p>
              <p className="text-sm text-muted-foreground mt-6">点击查看答案</p>
            </div>
          ) : (
            // 答案面
            <div className="text-center">
              <p className="text-sm text-primary font-medium mb-4">答案</p>
              <p className="text-lg leading-relaxed">{answer}</p>
            </div>
          )}
        </div>
      </div>

      {/* 底部操作栏 */}
      <div className="mobile-px pb-6 pt-2 space-y-3">
        {isFlipped && (
          <>
            {/* 下一张按钮 */}
            <MobileTouchButton
              variant="primary"
              onClick={onNext}
              className="w-full flex items-center justify-center gap-2"
            >
              <span>下一张</span>
              <ChevronRight className="w-5 h-5" />
            </MobileTouchButton>

            {/* 操作按钮组 */}
            <div className="flex gap-3">
              <MobileTouchButton
                variant="ghost"
                onClick={onSkip}
                className="flex-1"
              >
                跳过
              </MobileTouchButton>

              {onAiExplain && (
                <MobileTouchButton
                  variant="secondary"
                  onClick={onAiExplain}
                  className="flex-1 flex items-center justify-center gap-2"
                >
                  <Lightbulb className="w-4 h-4" />
                  <span>AI 解释</span>
                </MobileTouchButton>
              )}
            </div>
          </>
        )}
      </div>
    </div>
  )
}
```

## 功能说明

### MobileFreestylePage（主页面）
移动端随心模式的入口页面。

**主要功能：**
1. **今日进度展示**：显示已完成/总数、进度条
2. **开始练习**：大号触控按钮，一键开始
3. **统计卡片**：已完成、待练习数量
4. **功能说明**：向用户介绍随心模式
5. **设置面板**：简化的配置（底部抽屉）

**交互优化：**
- 大号触控按钮（满足 44px 最小尺寸）
- 卡片式布局，清晰的视觉层次
- 加载状态和空状态处理

### MobileFreestyleCard（练习卡片）
单张卡片的展示和交互。

**功能：**
- 点击翻转（问题 → 答案）
- 下一张、跳过按钮
- AI 解释（可选）

**后续扩展：**
- 滑动手势切换卡片
- 翻转动画效果
- 音效反馈

## 与桌面端的差异

| 特性 | 桌面端 | 移动端 |
|------|--------|--------|
| 布局 | 侧边栏 + 主内容区 | 全屏单列 |
| 配置 | 详细的配置面板 | 简化设置（链接到桌面端） |
| 卡片展示 | 中等尺寸 | 全屏大卡片 |
| 交互方式 | 鼠标点击 | 触控 + 手势 |
| AI 解释 | 侧边面板 | 底部抽屉 |
| 进度展示 | 详细图表 | 简化进度条 + 数字 |

## 数据流

```
加载配置和进度
       ↓
显示主页面（进度、统计）
       ↓
点击"开始练习"
       ↓
构建练习队列
       ↓
显示卡片（MobileFreestyleCard）
       ↓
完成/跳过 → 更新进度
       ↓
下一张卡片 / 完成练习
```
