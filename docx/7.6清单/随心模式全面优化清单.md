## A. 音效修复

### 背景与根本原因

`legacyWebAudio.ts` 使用一个全局共享的 `AudioContext`（`sharedAudioContext`）。在移动端浏览器（尤其是 iOS Safari 和 Android Chrome PWA 模式），`AudioContext` 在创建后处于 `suspended` 状态，必须在用户手势的同步调用栈内调用 `resume()` 才能激活。

当前代码在 `playToneSequence()` 里调用了 `context.resume()`，但这是在异步的播放逻辑里触发的，并不处于用户手势的同步调用栈内，导致 iOS 上的 `resume()` 被忽略，声音始终不响。

### A-1. 修复 `legacyWebAudio.ts` — 添加用户手势自动解锁逻辑

**文件**：`apps/web/src/shared/components/mindmap-host/legacyWebAudio.ts`

**修改位置**：在 `sharedAudioContext` 变量声明之后（大约第 5 行），插入以下代码块：

```typescript
// ─── 移动端 AudioContext 自动解锁 ───────────────────────────────────────────
// iOS Safari / Android Chrome PWA 要求 AudioContext 必须在用户手势的
// 同步调用栈内调用 resume()，否则永久 suspended。
// 此处监听全局用户交互事件，第一次触摸/点击时主动创建并 resume AudioContext。
function _unlockAudioContext() {
  let ctx = sharedAudioContext
  if (!ctx) {
    const Ctor = resolveAudioContextConstructor()
    if (Ctor) {
      ctx = new Ctor()
      sharedAudioContext = ctx
    }
  }
  if (ctx && ctx.state !== 'running') {
    void ctx.resume().catch(() => undefined)
  }
}

if (typeof document !== 'undefined') {
  const _unlockEvents = ['touchstart', 'pointerdown', 'click'] as const
  const _handleUnlock = () => {
    _unlockAudioContext()
    // 解锁后不需要继续监听（但不移除，因为页面回到后台再回来会再次 suspended）
  }
  for (const evt of _unlockEvents) {
    document.addEventListener(evt, _handleUnlock, { passive: true, capture: true })
  }
}
// ────────────────────────────────────────────────────────────────────────────
```

**注意**：这段代码要放在 `resolveAudioContextConstructor` 函数定义之后、`getSharedAudioContext` 函数定义之前，因为它调用了 `resolveAudioContextConstructor`。

### B-1. 后端 — 新建讲解服务文件

**新建**：`apps/api/src/memory_anki/modules/palace_quiz/application/quiz_explain_question.py`

```python
"""通用题目 AI 讲解服务。"""
from __future__ import annotations
from sqlalchemy.orm import Session
from memory_anki.modules.settings.application.ai_model_registry import AiRuntimeOptions
import json


def _ai_service():
    from . import ai_service
    return ai_service


def _build_question_context(session: Session, question_id: int) -> tuple[object, str]:
    from memory_anki.modules.palace_quiz.application.question_lookup_queries import get_question_by_id
    q = get_question_by_id(session, question_id=question_id)
    if q is None:
        raise ValueError(f"题目 {question_id} 不存在")

    lines = [f"题型：{q.question_type}", f"题干：{q.stem}"]
    if q.options_json:
        try:
            opts = json.loads(q.options_json) if isinstance(q.options_json, str) else q.options_json
            for opt in (opts or []):
                lines.append(f"  选项 {opt.get('label','')}：{opt.get('text','')}")
        except Exception:
            pass
    if q.answer_payload_json:
        try:
            ans = json.loads(q.answer_payload_json) if isinstance(q.answer_payload_json, str) else q.answer_payload_json
            lines.append(f"正确答案：{ans}")
        except Exception:
            pass
    if q.palace_id:
        from memory_anki.infrastructure.db.models import Palace
        palace = session.get(Palace, q.palace_id)
        if palace:
            lines.append(f"所属宫殿：{palace.title}")

    return q, "\n".join(lines)


def explain_question(
    session: Session,
    *,
    question_id: int,
    user_question: str,
    ai_options: AiRuntimeOptions | None = None,
) -> dict[str, object]:
    q, context = _build_question_context(session, question_id)

    messages = [
        {
            "role": "system",
            "content": (
                "你是一名专业的学习辅导老师，善于用简洁清晰的语言讲解知识点。"
                "用户正在做一道记忆宫殿学习题，请根据题目内容回答用户的问题。"
                "回答不超过 300 字，语言自然，可适当用列表或加粗关键词。"
            ),
        },
        {
            "role": "user",
            "content": f"{context}\n\n---\n\n用户的问题：{user_question}",
        },
    ]

    response_text, log_id = _ai_service()._call_logged_chat_completion(
        config=ai_options,
        extra_payload={},
        feature="随心模式",
        operation="freestyle_explain_question",
        palace_id=getattr(q, "palace_id", None),
        messages=messages,
        response_format=None,
        request_payload={"question_id": question_id, "user_question": user_question},
    )

    return {
        "question_id": question_id,
        "explanation_text": response_text.strip(),
        "ai_call_log_id": log_id,
        "resolved_ai": getattr(ai_options, "model_id", None),
    }
```

class QuestionExplainRequest(BaseModel):

    explanation_text: str

@router.post("/quiz-questions/{question_id}/explain", response_model=QuestionExplainResponse)
def explain_question_endpoint(

    body: QuestionExplainRequest,
    db: Session = Depends(get_db),

```tsx
import { useState } from 'react'
import { X, Send, Sparkles } from 'lucide-react'
import { Button } from '@/shared/components/ui/button'
import { requestQuestionExplainApi } from '@/features/palace-quiz/api'
import type { FreestyleQuizCard } from '@/shared/api/contracts'

const PRESETS = [
  { label: '解释考点', question: '请解释这道题的核心考点是什么。' },
  { label: '记忆技巧', question: '给我一个记住这道题答案的技巧或口诀。' },
  { label: '为什么选这个', question: '为什么正确答案是这个？其他选项错在哪里？' },
  { label: '延伸思考', question: '这个知识点还能延伸出哪些相关考点？' },
]

interface Props {
  open: boolean
  card: FreestyleQuizCard | null
  onClose: () => void
}

export function FreestyleAiExplainSheet({ open, card, onClose }: Props) {
  const [input, setInput] = useState('')
  const [loading, setLoading] = useState(false)
  const [result, setResult] = useState<string | null>(null)
  const [error, setError] = useState<string | null>(null)

  async function send(question: string) {
    if (!card || !question.trim() || loading) return
    setLoading(true)
    setResult(null)
    setError(null)
    try {
      const res = await requestQuestionExplainApi(card.question.id, question.trim())
      setResult(res.explanation_text)
    } catch (e) {
      setError(e instanceof Error ? e.message : 'AI 讲解请求失败')
    } finally {
      setLoading(false)
    }
  }

  if (!open) return null

  return (
    <div className="fixed inset-0 z-50 flex flex-col justify-end">
      <div className="absolute inset-0 bg-black/60 backdrop-blur-sm" onClick={onClose} />
      <div className="relative z-10 flex max-h-[82vh] flex-col rounded-t-2xl border-t border-white/10 bg-zinc-900 shadow-2xl">
        <div className="mx-auto mt-3 h-1 w-10 rounded-full bg-zinc-700" />
        <div className="flex items-center justify-between px-5 py-3">
          <div className="flex items-center gap-2">
            <Sparkles className="size-4 text-amber-400" />
            <span className="text-sm font-semibold text-zinc-100">AI 讲解</span>
          </div>
          <button type="button" onClick={onClose}
            className="rounded-full p-1.5 text-zinc-400 hover:bg-zinc-800 hover:text-zinc-100">
            <X className="size-4" />
          </button>
        </div>
        <div className="flex gap-2 overflow-x-auto px-5 pb-3 [scrollbar-width:none]">
          {PRESETS.map((p) => (
            <button key={p.label} type="button" disabled={loading}
              onClick={() => { setInput(p.question); void send(p.question) }}
              className="shrink-0 rounded-full border border-white/10 bg-zinc-800 px-3 py-1.5 text-xs
                         text-zinc-300 transition-all hover:bg-zinc-700 hover:text-zinc-50
                         active:scale-95 disabled:opacity-40">
              {p.label}
            </button>
          ))}
        </div>
        <div className="flex items-end gap-2 border-t border-white/8 px-4 py-3">
          <textarea rows={2}
            className="min-h-[52px] flex-1 resize-none rounded-xl border border-white/10 bg-zinc-800
                       px-3 py-2 text-sm text-zinc-100 placeholder:text-zinc-500
                       focus:outline-none focus:ring-1 focus:ring-amber-500/60"
            placeholder="或者直接输入你的问题…"
            value={input}
            onChange={(e) => setInput(e.target.value)}
          />
          <Button type="button" size="icon" disabled={loading || !input.trim()}
            onClick={() => void send(input)}
            className="shrink-0 rounded-xl bg-amber-500 text-zinc-950 hover:bg-amber-400">
            <Send className="size-4" />
          </Button>
        </div>
        {(loading || result || error) && (
          <div className="min-h-0 flex-1 overflow-y-auto border-t border-white/8 px-5 py-4">
            {loading && (
              <div className="flex items-center gap-2 text-sm text-zinc-400">
                <span className="inline-block size-3 animate-spin rounded-full border-2 border-amber-400 border-t-transparent" />
                AI 正在思考…
              </div>
            )}
            {result && <p className="whitespace-pre-wrap text-sm leading-relaxed text-zinc-200">{result}</p>}
            {error && <p className="text-sm text-red-400">{error}</p>}
          </div>
        )}
        <div className="h-[env(safe-area-inset-bottom,0px)]" />
      </div>
    </div>
  )
}
```
