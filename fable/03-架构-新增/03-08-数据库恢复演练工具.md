# 03-08 数据库恢复演练工具

## 1. 原始需求

`modules/backups/application/backup_lifecycle.py` 提供了完善的备份能力（启动/停机/周期/滚动备份，`FULL_BACKUPS_DIR` 保留最近 8 份），但**从未验证过备份真的可恢复**：`restore_database_backup`（第 120~132 行）只在真实恢复时才执行，没有演练机制。若备份因百度网盘同步冲突、复制中断等原因损坏，只会在灾难当天发现。目标：新增 `tools/verify_backup.py`——取最近一份含数据库的备份，在临时目录"恢复"，做 SQLite 完整性检查、alembic 版本核对、关键表行数与当前生产库对比，输出报告。脚本风格参照既有 `tools/create_startup_backup.py`（sys.path 注入 + 复用 `memory_anki` 包能力，第 14~23 行）。

## 2. 详细执行清单

### 步骤 1：新建 `tools/verify_backup.py`

完整文件内容如下（新文件，约 150 行）：

```python
"""备份恢复演练工具。

用法：python tools/verify_backup.py [备份目录路径]
不传参数时自动选取 FULL_BACKUPS_DIR 下最近一份含数据库的备份。

流程：复制备份数据库到临时目录 → PRAGMA integrity_check →
alembic_version 与生产库核对 → 关键表行数对比 → 输出报告，
并把 JSON 报告写入 APP_HOME/backup-verify-reports/。
只读操作：绝不改写生产库与备份目录。
"""
from __future__ import annotations

import json
import shutil
import sqlite3
import sys
import tempfile
from datetime import datetime
from pathlib import Path

_REPO_ROOT = Path(__file__).resolve().parents[1]
_API_SRC = _REPO_ROOT / "apps" / "api" / "src"
if _API_SRC.is_dir():
    sys.path.insert(0, str(_API_SRC))

from memory_anki.core.config import APP_HOME, DB_PATH, FULL_BACKUPS_DIR  # noqa: E402

KEY_TABLES = [
    "palaces",
    "pegs",
    "palace_segments",
    "palace_quiz_questions",
    "review_schedules",
    "review_logs",
    "study_sessions",
    "subjects",
    "chapters",
    "config",
]

REPORTS_DIR = APP_HOME / "backup-verify-reports"


def find_latest_backup_with_db() -> Path | None:
    if not FULL_BACKUPS_DIR.exists():
        return None
    candidates = [
        folder
        for folder in FULL_BACKUPS_DIR.iterdir()
        if folder.is_dir() and (folder / DB_PATH.name).exists()
    ]
    if not candidates:
        return None
    return max(candidates, key=lambda item: item.stat().st_mtime)


def read_alembic_version(db_file: Path) -> str | None:
    try:
        with sqlite3.connect(db_file) as conn:
            row = conn.execute("SELECT version_num FROM alembic_version").fetchone()
            return row[0] if row else None
    except sqlite3.Error:
        return None


def integrity_check(db_file: Path) -> str:
    with sqlite3.connect(db_file) as conn:
        row = conn.execute("PRAGMA integrity_check").fetchone()
        return str(row[0]) if row else "no result"


def table_counts(db_file: Path) -> dict[str, int]:
    counts: dict[str, int] = {}
    with sqlite3.connect(db_file) as conn:
        for table in KEY_TABLES:
            try:
                row = conn.execute(f"SELECT COUNT(*) FROM {table}").fetchone()
                counts[table] = int(row[0])
            except sqlite3.Error:
                counts[table] = -1  # 表不存在
    return counts


def main() -> int:
    if len(sys.argv) > 1:
        backup_dir = Path(sys.argv[1])
    else:
        backup_dir = find_latest_backup_with_db()
    if backup_dir is None or not backup_dir.is_dir():
        print("[FAIL] 未找到含数据库的备份目录。")
        return 2
    backup_db = backup_dir / DB_PATH.name
    if not backup_db.exists():
        print(f"[FAIL] 备份中缺少数据库文件：{backup_db}")
        return 2

    temp_dir = Path(tempfile.mkdtemp(prefix="memory-anki-verify-"))
    report: dict = {
        "backup_dir": str(backup_dir),
        "verified_at": datetime.now().isoformat(timespec="seconds"),
        "ok": True,
        "problems": [],
    }
    try:
        restored_db = temp_dir / DB_PATH.name
        shutil.copy2(backup_db, restored_db)

        integrity = integrity_check(restored_db)
        report["integrity_check"] = integrity
        if integrity != "ok":
            report["ok"] = False
            report["problems"].append(f"integrity_check 失败：{integrity}")

        backup_version = read_alembic_version(restored_db)
        live_version = read_alembic_version(DB_PATH) if DB_PATH.exists() else None
        report["alembic_version"] = {"backup": backup_version, "live": live_version}
        if backup_version is None:
            report["ok"] = False
            report["problems"].append("备份库缺少 alembic_version 表。")
        elif live_version and backup_version != live_version:
            report["problems"].append(
                f"备份迁移版本 {backup_version} 落后于生产 {live_version}（恢复后需 alembic upgrade head）。"
            )

        backup_counts = table_counts(restored_db)
        live_counts = table_counts(DB_PATH) if DB_PATH.exists() else {}
        report["table_counts"] = {"backup": backup_counts, "live": live_counts}
        for table, count in backup_counts.items():
            if count == -1 and live_counts.get(table, -1) != -1:
                report["ok"] = False
                report["problems"].append(f"备份库缺少表 {table}。")
            live = live_counts.get(table)
            if live is not None and live >= 0 and count >= 0 and count < live:
                # 备份必然略旧，行数少于生产属正常，仅提示不判失败。
                report["problems"].append(
                    f"表 {table}：备份 {count} 行 < 生产 {live} 行（备份时点差异）。"
                )
    finally:
        shutil.rmtree(temp_dir, ignore_errors=True)

    REPORTS_DIR.mkdir(parents=True, exist_ok=True)
    report_path = REPORTS_DIR / f"verify-{datetime.now().strftime('%Y%m%d-%H%M%S')}.json"
    report_path.write_text(json.dumps(report, ensure_ascii=False, indent=2), encoding="utf-8")

    status = "PASS" if report["ok"] else "FAIL"
    print(f"[{status}] 备份：{backup_dir.name}")
    print(f"  integrity_check: {report.get('integrity_check')}")
    print(f"  alembic_version: {report.get('alembic_version')}")
    for problem in report["problems"]:
        print(f"  - {problem}")
    print(f"  报告已写入: {report_path}")
    return 0 if report["ok"] else 1


if __name__ == "__main__":
    raise SystemExit(main())
```

要点说明（执行者不要自行更改这些决策）：

- **只读演练**：只 `copy2` 备份库到临时目录检查，绝不调用 `restore_storage_backup`，绝不写生产目录；
- 不 import `backup_lifecycle`（其 `_latest_full_backup` 是私有函数且不筛"含数据库"），脚本自带查找逻辑；
- 行数"备份 < 生产"只提示不判失败（备份天然滞后）；表缺失、integrity 失败、无 alembic_version 才判 FAIL；
- 报告目录 `APP_HOME/backup-verify-reports/` 经 `memory_anki.core.config` 派生，符合跨设备约束。

**自查点**：`python tools/verify_backup.py`（存在备份时）打印 `[PASS]` 并生成报告文件；`python tools/verify_backup.py D:\不存在的路径` 返回码 2。

### 步骤 2：单元测试

新建 `apps/api/tests/test_verify_backup_tool.py`：

```python
import sqlite3
import subprocess
import sys
from pathlib import Path

REPO_ROOT = Path(__file__).resolve().parents[3]
SCRIPT = REPO_ROOT / "tools" / "verify_backup.py"


def _make_fake_backup(tmp_path: Path, db_name: str) -> Path:
    backup_dir = tmp_path / "20260101-000000-test"
    backup_dir.mkdir(parents=True)
    db_file = backup_dir / db_name
    with sqlite3.connect(db_file) as conn:
        conn.execute("CREATE TABLE alembic_version (version_num TEXT)")
        conn.execute("INSERT INTO alembic_version VALUES ('0012_freestyle_history')")
        conn.execute("CREATE TABLE palaces (id INTEGER PRIMARY KEY, title TEXT)")
        conn.execute("INSERT INTO palaces (title) VALUES ('t1')")
        conn.commit()
    return backup_dir


def test_verify_backup_pass_on_valid_backup(tmp_path, monkeypatch):
    monkeypatch.setenv("MEMORY_ANKI_HOME", str(tmp_path / "home"))
    backup_dir = _make_fake_backup(tmp_path, "memory_palace.db")
    result = subprocess.run(
        [sys.executable, str(SCRIPT), str(backup_dir)],
        capture_output=True,
        text=True,
    )
    assert result.returncode == 0, result.stdout + result.stderr
    assert "[PASS]" in result.stdout


def test_verify_backup_fails_without_db(tmp_path, monkeypatch):
    monkeypatch.setenv("MEMORY_ANKI_HOME", str(tmp_path / "home"))
    empty_dir = tmp_path / "empty-backup"
    empty_dir.mkdir()
    result = subprocess.run(
        [sys.executable, str(SCRIPT), str(empty_dir)],
        capture_output=True,
        text=True,
    )
    assert result.returncode == 2
```

说明：用子进程跑脚本并以 `MEMORY_ANKI_HOME` 指向临时目录，使 `core/config.py` 的路径派生（第 102~111 行 `_resolve_app_home`）落在沙箱里，不触碰真实运行时数据。数据库文件名 `memory_palace.db` 与 `DB_PATH.name` 一致（config.py 第 130 行）。

**自查点**：`cd apps/api && python -m pytest tests/test_verify_backup_tool.py -q` 全绿。

## 3. 测试验收标准

可执行命令：

- `cd apps/api && python -m pytest tests/test_verify_backup_tool.py -q` → 全绿。
- `cd apps/api && python -m pytest` → 无新增失败（本工具不触及生产代码）。
- 真机验证：`python tools/verify_backup.py` → 打印 `[PASS]`，`%LOCALAPPDATA%\MemoryAnki\backup-verify-reports\`（或 MEMORY_ANKI_HOME 对应目录）出现 JSON 报告。

行为验收：

1. 先 `POST /api/v1/backups/create` 生成新备份，再跑脚本 → 选中的是刚才的备份（输出目录名可核对）。
2. 手工把某备份里的 `memory_palace.db` 截断几个字节（复制一份备份目录来做，勿破坏真备份）后指定路径跑脚本 → `[FAIL]` 且 integrity_check 非 ok，返回码 1。
3. 传入不存在路径 → 返回码 2，提示未找到。

回归检查：

- 跑完演练后生产库 `memory_palace.db` 的 mtime/大小不变（只读保证）；
- 备份目录内容不变；
- 服务运行中执行脚本不影响在线请求（SQLite 只读连接备份副本）。
