# 06-02 学习热力图与连续打卡

## 1. 原始需求

数据早已齐备但没有任何长期可视化：

1. `review_logs` 表（`apps/api/src/memory_anki/infrastructure/db/_tables/palaces.py` 第 295-312 行）有 `review_date`、`duration_seconds`，且已有 `ix_review_logs_date` 单列索引；`study_sessions` 表（`_tables/misc.py` 第 15-69 行）有 `started_at`、`effective_seconds`、`deleted_at`。
2. 仪表盘（`apps/api/src/memory_anki/modules/dashboard/application/service.py`）目前只算"今日/本周/本月"三个时间窗，看不到历史节奏；前端 `apps/web/src/features/dashboard/DashboardPage.tsx` 已引入 recharts（`package.json` 中 `recharts: ^3.8.1`）画其他图表。

期望效果：仪表盘出现一块近 26 周（182 天）的 GitHub 风格热力图，每格颜色深浅代表当日学习强度；上方显示"当前连续 N 天 / 历史最长 M 天"。

**实现取舍**：GitHub 热力图本质是 7×26 的方格网格，recharts 没有现成 heatmap 组件，用 recharts 硬造反而复杂——直接用 Tailwind 的 CSS grid 渲染 `div` 方格即可（克制原则）。打卡定义：当日存在任意一条 `review_logs` 或任意一条有效 `study_sessions`（`deleted_at IS NULL` 且 `effective_seconds > 0`）即算打卡。

### 后端步骤

#### 步骤 1：新建热力图聚合服务

新建 `apps/api/src/memory_anki/modules/dashboard/application/heatmap_service.py`，完整内容：

```python
"""Daily study heatmap and streak aggregation."""

from __future__ import annotations

from datetime import date, datetime, time, timedelta

from sqlalchemy import func
from sqlalchemy.orm import Session

from memory_anki.infrastructure.db.models import ReviewLog, StudySession

MAX_HEATMAP_DAYS = 366
DEFAULT_HEATMAP_DAYS = 182


def build_heatmap_payload(session: Session, days: int = DEFAULT_HEATMAP_DAYS) -> dict:
    days = max(7, min(int(days), MAX_HEATMAP_DAYS))
    today = date.today()
    start = today - timedelta(days=days - 1)
    start_dt = datetime.combine(start, time.min)

    review_rows = (
        session.query(
            ReviewLog.review_date,
            func.count(ReviewLog.id),
            func.coalesce(func.sum(ReviewLog.duration_seconds), 0),
        )
        .filter(ReviewLog.review_date >= start, ReviewLog.review_date <= today)
        .group_by(ReviewLog.review_date)
        .all()
    )
    session_rows = (
        session.query(
            func.date(StudySession.started_at),
            func.coalesce(func.sum(StudySession.effective_seconds), 0),
        )
        .filter(
            StudySession.deleted_at.is_(None),
            StudySession.started_at >= start_dt,
            StudySession.effective_seconds > 0,
        )
        .group_by(func.date(StudySession.started_at))
        .all()
    )

    by_day: dict[str, dict] = {}
    for review_date, count, duration in review_rows:
        if review_date is None:
            continue
        key = review_date.isoformat()
        entry = by_day.setdefault(key, {"review_count": 0, "study_seconds": 0})
        entry["review_count"] = int(count)
        entry["study_seconds"] += int(duration)
    for day_value, seconds in session_rows:
        key = str(day_value)
        entry = by_day.setdefault(key, {"review_count": 0, "study_seconds": 0})
        entry["study_seconds"] += int(seconds)

    items = []
    active_days: set[str] = set()
    for offset in range(days):
        day = start + timedelta(days=offset)
        key = day.isoformat()
        entry = by_day.get(key, {"review_count": 0, "study_seconds": 0})
        active = entry["review_count"] > 0 or entry["study_seconds"] > 0
        if active:
            active_days.add(key)
        items.append(
            {
                "date": key,
                "review_count": entry["review_count"],
                "study_seconds": entry["study_seconds"],
                "active": active,
            }
        )

    current_streak = 0
    cursor = today
    # 今天还没学不打断连击：从今天或昨天开始向前数
    if today.isoformat() not in active_days:
        cursor = today - timedelta(days=1)
    while cursor >= start and cursor.isoformat() in active_days:
        current_streak += 1
        cursor -= timedelta(days=1)

    longest_streak = 0
    run = 0
    for item in items:
        run = run + 1 if item["active"] else 0
        longest_streak = max(longest_streak, run)

    return {
        "start_date": start.isoformat(),
        "end_date": today.isoformat(),
        "items": items,
        "current_streak": current_streak,
        "longest_streak": longest_streak,
        "active_day_count": len(active_days),
    }
```

注意：`longest_streak` 仅统计窗口内数据（182 天），这是有意的简化，不要为了"历史最长"扫全表。

自查点：`cd apps/api && python -c "from memory_anki.modules.dashboard.application.heatmap_service import build_heatmap_payload"` 无报错。

#### 步骤 4：热力图组件（FSD：features/dashboard 内部组件）

新建 `apps/web/src/features/dashboard/components/StudyHeatmap.tsx`（features/dashboard 目前没有 components 子目录，新建之），核心结构：

```tsx
import { useEffect, useState } from 'react'
import { Card, CardContent, CardHeader, CardTitle } from '@/shared/components/ui/card'
import { getDashboardHeatmapApi } from '../api'
import type { DashboardHeatmapResponse, HeatmapDayItem } from '@/shared/api/contracts'

function intensityClass(item: HeatmapDayItem) {
  if (!item.active) return 'bg-muted'
  const minutes = item.study_seconds / 60 + item.review_count * 5
  if (minutes >= 90) return 'bg-emerald-600'
  if (minutes >= 45) return 'bg-emerald-500'
  if (minutes >= 15) return 'bg-emerald-400'
  return 'bg-emerald-200'
}

export function StudyHeatmap() {
  const [data, setData] = useState<DashboardHeatmapResponse | null>(null)

  useEffect(() => {
    let cancelled = false
    getDashboardHeatmapApi()
      .then((payload) => {
        if (!cancelled) setData(payload)
      })
      .catch(() => {})
    return () => {
      cancelled = true
    }
  }, [])

  if (!data) return null
  // 组织为按周分列：grid-flow-col + grid-rows-7，每格 title 提示明细
  return (
    <Card>
      <CardHeader className="flex flex-row items-center justify-between">
        <CardTitle className="text-base">学习热力图</CardTitle>
        <div className="text-sm text-muted-foreground">
          连续 {data.current_streak} 天 · 最长 {data.longest_streak} 天 · 近半年学习 {data.active_day_count} 天
        </div>
      </CardHeader>
      <CardContent className="overflow-x-auto">
        <div className="grid grid-flow-col grid-rows-7 gap-1 w-max">
          {data.items.map((item) => (
            <div
              key={item.date}
              title={`${item.date}：复习 ${item.review_count} 次，学习 ${Math.round(item.study_seconds / 60)} 分钟`}
              className={`size-3 rounded-sm ${intensityClass(item)}`}
            />
          ))}
        </div>
      </CardContent>
    </Card>
  )
}
```

实现要点（执行者需补齐）：为了让每列从周一开始对齐，在 `items` 前补 `startDate.getDay()` 相应数量的空占位格（`<div className="size-3" />`）。`Card` 组件路径以 `DashboardPage.tsx` 顶部现有 import 为准（该页已使用同一套 ui 组件）。

**不要**用 recharts 实现该网格；**不要**把组件放到 `shared/components`（它依赖 feature 的 API，属于 dashboard feature 内部件）。
