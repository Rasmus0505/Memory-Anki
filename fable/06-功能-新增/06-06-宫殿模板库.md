# 06-06 宫殿模板库

## 1. 原始需求

现状：记忆宫殿的桩点结构高度可复用（身体部位桩、房间桩、路线桩），但每次新建宫殿（`/palaces/new`，对应 `apps/web/src/features/palace-edit/PalaceEditPage.tsx`）都要从空白思维导图手搭一遍。宫殿的结构数据存在 `palaces.editor_doc`（Text，思维导图 JSON）与 `editor_config` 两列（`apps/api/src/memory_anki/infrastructure/db/_tables/palaces.py` 第 70~71 行），编辑接口为 `GET/PUT /api/v1/palaces/{id}/editor`（`modules/palaces/presentation/router.py` 第 218、250 行）。

期望：新增 `palace_templates` 表（alembic 迁移）保存 `editor_doc` 快照；提供模板增删查与"从模板实例化宫殿"API；前端在 `PalaceEditPage` 上做两件事——编辑已有宫殿时可"存为模板"，`/palaces/new` 空白新建时弹出模板选择（含"空白宫殿"选项）。

已核实的衔接点：
- 创建宫殿走 `POST /api/v1/palaces`（`PalaceCreate` schema 只有 title/description/pegs，见 `modules/palaces/domain/schemas.py` 第 17~20 行），创建后 `trigger_review_for_palace` 生成复习计划（router 第 155~160 行）；
- alembic 当前 HEAD 为 `0012_freestyle_history`（`apps/api/alembic/versions/`），新迁移编号 0013，风格参照 `0012_freestyle_history.py`（`_table_exists`/`_create_index_once` 守卫写法）。

本期不做：
- 不做模板市场/内置模板包，模板只来自用户自己的宫殿；
- 不做模板编辑（要改就删掉重存）；
- 不复用 `palace_versions` 表当模板（版本快照会随宫殿删除级联清掉，语义不符），新建独立表；
- 不迁移 `pegs` 树（编辑器以 `editor_doc` 为准，pegs 由既有同步机制维护）。

3. 新建 `apps/api/src/memory_anki/modules/palaces/application/palace_template_service.py`，完整内容：

```python
"""Palace template CRUD and instantiation."""

from __future__ import annotations

import json

from sqlalchemy.orm import Session

from memory_anki.infrastructure.db.models import Palace, PalaceTemplate


class PalaceTemplateError(ValueError):
    pass


def _template_json(template: PalaceTemplate) -> dict:
    return {
        "id": template.id,
        "name": template.name,
        "description": template.description,
        "source_palace_id": template.source_palace_id,
        "created_at": template.created_at.isoformat() if template.created_at else None,
    }


def list_templates(session: Session) -> list[dict]:
    rows = session.query(PalaceTemplate).order_by(PalaceTemplate.id.desc()).all()
    return [_template_json(row) for row in rows]


def create_template_from_palace(session: Session, palace_id: int, name: str, description: str) -> dict:
    palace = session.get(Palace, palace_id)
    if not palace:
        raise PalaceTemplateError("宫殿不存在。")
    if not str(palace.editor_doc or "").strip():
        raise PalaceTemplateError("这个宫殿还没有思维导图内容，无法存为模板。")
    template = PalaceTemplate(
        name=str(name or "").strip() or (palace.manual_title or palace.title or "未命名模板"),
        description=str(description or "").strip(),
        editor_doc=palace.editor_doc or "",
        editor_config=palace.editor_config or "",
        source_palace_id=palace.id,
    )
    session.add(template)
    session.commit()
    session.refresh(template)
    return _template_json(template)


def delete_template(session: Session, template_id: int) -> bool:
    template = session.get(PalaceTemplate, template_id)
    if not template:
        return False
    session.delete(template)
    session.commit()
    return True


def _retitle_editor_doc(editor_doc: str, title: str) -> str:
    """把模板 editor_doc 根节点文本替换为新宫殿标题；解析失败则原样返回。"""
    try:
        doc = json.loads(editor_doc)
    except (TypeError, ValueError):
        return editor_doc
    node_data = doc.get("nodeData") if isinstance(doc, dict) else None
    if isinstance(node_data, dict) and title:
        node_data["topic"] = title
    return json.dumps(doc, ensure_ascii=False)


def instantiate_template(session: Session, template_id: int, title: str) -> Palace:
    template = session.get(PalaceTemplate, template_id)
    if not template:
        raise PalaceTemplateError("模板不存在。")
    safe_title = str(title or "").strip() or template.name
    palace = Palace(
        title=safe_title,
        description=f"由模板「{template.name}」创建",
        editor_doc=_retitle_editor_doc(template.editor_doc, safe_title),
        editor_config=template.editor_config or "",
    )
    session.add(palace)
    session.commit()
    session.refresh(palace)
    return palace
```

   - 注意：执行前先打开一份真实 `palaces.editor_doc` 确认根节点字段名（`nodeData.topic` 是 mind-elixir 风格；若实际是别的结构，按实际字段改 `_retitle_editor_doc`，改不动就直接返回原文档——标题不同步不算失败）。
   - 自查点：import 无报错。

4. 在 `apps/api/src/memory_anki/modules/palaces/presentation/router.py` 注册路由（放在 `/palaces/{palace_id}/versions` 系列附近；注意静态路径 `/palace-templates` 与既有动态路由无冲突）：
   - 顶部 import：`from memory_anki.modules.palaces.application.palace_template_service import (PalaceTemplateError, create_template_from_palace, delete_template, instantiate_template, list_templates)`；`trigger_review_for_palace` 与 `palace_json` 已有 import。

```python
@router.get("/palace-templates")
def api_list_palace_templates(s: Session = Depends(session_dep)):
    return {"items": list_templates(s)}


@router.post("/palace-templates")
def api_create_palace_template(data: dict, s: Session = Depends(session_dep)):
    try:
        item = create_template_from_palace(
            s,
            int(data.get("palace_id") or 0),
            str(data.get("name") or ""),
            str(data.get("description") or ""),
        )
    except PalaceTemplateError as exc:
        raise HTTPException(status_code=400, detail=str(exc)) from exc
    return {"item": item}


@router.delete("/palace-templates/{template_id}")
def api_delete_palace_template(template_id: int, s: Session = Depends(session_dep)):
    return {"ok": delete_template(s, template_id)}


@router.post("/palace-templates/{template_id}/instantiate")
def api_instantiate_palace_template(template_id: int, data: dict, s: Session = Depends(session_dep)):
    try:
        palace = instantiate_template(s, template_id, str(data.get("title") or ""))
    except PalaceTemplateError as exc:
        raise HTTPException(status_code=400, detail=str(exc)) from exc
    trigger_review_for_palace(s, palace.id)
    return palace_json(palace, s)
```

5. 在 `apps/web/src/entities/palace/api/`（`getPalaceEditorApi` 所在文件）追加：

```ts
export interface PalaceTemplateSummary {
  id: number
  name: string
  description: string
  source_palace_id: number | null
  created_at: string | null
}

export function listPalaceTemplatesApi() {
  return request<{ items: PalaceTemplateSummary[] }>('/palace-templates')
}

export function createPalaceTemplateApi(data: { palace_id: number; name: string; description?: string }) {
  return request<{ item: PalaceTemplateSummary }>('/palace-templates', {
    method: 'POST',
    body: JSON.stringify(data),
    persistence: { resourceKey: 'palace-template:create', description: '存为宫殿模板', replayMode: 'manual' },
  })
}

export function deletePalaceTemplateApi(id: number) {
  return request<{ ok: boolean }>(`/palace-templates/${id}`, {
    method: 'DELETE',
    persistence: { resourceKey: `palace-template:delete:${id}`, description: '删除宫殿模板', replayMode: 'manual' },
  })
}

export function instantiatePalaceTemplateApi(id: number, title: string) {
  return request<{ id: number }>(`/palace-templates/${id}/instantiate`, {
    method: 'POST',
    body: JSON.stringify({ title }),
    persistence: { resourceKey: 'palace-template:instantiate', description: '从模板创建宫殿', replayMode: 'manual' },
  })
}
```
