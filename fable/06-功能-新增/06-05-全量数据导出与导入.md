# 06-05 全量数据导出与导入

## 1. 原始需求

用户在两台 Windows 电脑间迁移数据目前只能手动拷数据库文件：

1. 现有导出是"宫殿级"的：`GET /api/v1/export/json` 与 `/export/markdown`（`apps/api/src/memory_anki/modules/palaces/presentation/import_router.py` 第 74-83 行）只导出宫殿/pegs，不含复习日程、做题记录、英语阅读、学习会话等其余表，也不含附件文件。
2. backups 模块已有整库备份基建：`create_full_backup`/`restore_database_backup`/`create_rescue_snapshot`/`list_backups`（`apps/api/src/memory_anki/modules/backups/application/backup_service.py` 门面导出，落盘目录 `FULL_BACKUPS_DIR` 派生自 `memory_anki.core.config`），并已在 `modules/palaces/presentation/router.py` 第 528-534 行暴露 `/backups`、`/backups/create`。但备份文件躺在本机 `APP_HOME` 里，没有"下载到浏览器 → 拷到另一台机器 → 上传导入"的通路。
3. 附件在 `ATTACHMENTS_DIR`（`core/config.py` 第 113 行），DB 在 `DB_PATH`（第 130 行），共 29 张表。

期望效果：设置页备份区一键下载 `memory-anki-full-YYYYMMDD.zip`（含所有表 JSON + 附件 + 清单），在另一台设备上传该 zip 完成整库导入。

**合并策略取舍（本文档结论：替换式导入 + 导入前自动逃生快照）**：逐行合并需要为 29 张表定义主键冲突、外键重映射与语义级去重规则，属于企业级同步引擎的复杂度，且个人双设备场景是"以一台为准搬到另一台"，不存在双向并发编辑合并的真实需求。因此导入 = 先 `create_rescue_snapshot()` 自动备份当前库，再整库清空重建。误导入可用既有备份恢复功能回滚。

### 后端步骤

#### 步骤 1：全量导出/导入服务

新建 `apps/api/src/memory_anki/modules/backups/application/full_transfer_service.py`，完整内容：

```python
"""Whole-database export/import as a portable zip (all tables JSON + attachments)."""

from __future__ import annotations

import io
import json
import zipfile
from datetime import date, datetime
from pathlib import Path
from typing import Any

from sqlalchemy import text
from sqlalchemy.orm import Session

from memory_anki.core.config import ATTACHMENTS_DIR
from memory_anki.core.time import utc_now_naive
from memory_anki.infrastructure.db.models import Base, engine

ARCHIVE_FORMAT_VERSION = 1
DATA_JSON_NAME = "data.json"
MANIFEST_JSON_NAME = "manifest.json"
ATTACHMENTS_PREFIX = "attachments/"


class FullTransferError(ValueError):
    pass


def _current_alembic_revision(session: Session) -> str:
    try:
        row = session.execute(text("SELECT version_num FROM alembic_version")).fetchone()
        return str(row[0]) if row else ""
    except Exception:
        return ""


def _json_default(value: Any):
    if isinstance(value, (datetime, date)):
        return value.isoformat()
    return str(value)


def _dump_all_tables(session: Session) -> dict[str, list[dict]]:
    result: dict[str, list[dict]] = {}
    for table in Base.metadata.sorted_tables:
        rows = session.execute(table.select()).mappings().all()
        result[table.name] = [dict(row) for row in rows]
    return result


def build_full_archive(session: Session) -> tuple[bytes, str]:
    """返回 (zip 字节, 建议文件名)。"""
    data = _dump_all_tables(session)
    manifest = {
        "format_version": ARCHIVE_FORMAT_VERSION,
        "alembic_revision": _current_alembic_revision(session),
        "created_at": utc_now_naive().isoformat(timespec="seconds"),
        "table_counts": {name: len(rows) for name, rows in data.items()},
    }
    buffer = io.BytesIO()
    with zipfile.ZipFile(buffer, "w", compression=zipfile.ZIP_DEFLATED) as archive:
        archive.writestr(MANIFEST_JSON_NAME, json.dumps(manifest, ensure_ascii=False, indent=2))
        archive.writestr(
            DATA_JSON_NAME,
            json.dumps(data, ensure_ascii=False, default=_json_default),
        )
        attachments_root = Path(ATTACHMENTS_DIR)
        if attachments_root.exists():
            for file_path in attachments_root.rglob("*"):
                if file_path.is_file():
                    relative = file_path.relative_to(attachments_root).as_posix()
                    archive.write(file_path, ATTACHMENTS_PREFIX + relative)
    filename = f"memory-anki-full-{date.today().strftime('%Y%m%d')}.zip"
    return buffer.getvalue(), filename


def inspect_archive(zip_bytes: bytes, session: Session) -> dict:
    """dry-run：校验 zip 并返回导入预览，不写库。"""
    try:
        archive = zipfile.ZipFile(io.BytesIO(zip_bytes))
    except zipfile.BadZipFile as exc:
        raise FullTransferError("上传的文件不是有效的 zip。") from exc
    names = set(archive.namelist())
    if MANIFEST_JSON_NAME not in names or DATA_JSON_NAME not in names:
        raise FullTransferError("zip 内缺少 manifest.json 或 data.json，不是全量导出包。")
    manifest = json.loads(archive.read(MANIFEST_JSON_NAME))
    if int(manifest.get("format_version") or 0) != ARCHIVE_FORMAT_VERSION:
        raise FullTransferError("导出包格式版本不兼容。")
    current_revision = _current_alembic_revision(session)
    archive_revision = str(manifest.get("alembic_revision") or "")
    attachment_count = sum(1 for name in names if name.startswith(ATTACHMENTS_PREFIX) and not name.endswith("/"))
    return {
        "manifest": manifest,
        "attachment_count": attachment_count,
        "schema_match": bool(current_revision) and archive_revision == current_revision,
        "current_alembic_revision": current_revision,
    }


def import_full_archive(zip_bytes: bytes, session: Session) -> dict:
    """替换式导入：清空所有业务表后按导出内容重建，并覆盖附件目录。"""
    preview = inspect_archive(zip_bytes, session)
    if not preview["schema_match"]:
        raise FullTransferError(
            "导出包的数据库版本与当前程序不一致，请先把两台设备升级到同一版本再迁移。"
        )
    archive = zipfile.ZipFile(io.BytesIO(zip_bytes))
    data: dict[str, list[dict]] = json.loads(archive.read(DATA_JSON_NAME))

    session.close()  # 释放当前会话，用独立连接做整库替换
    imported_counts: dict[str, int] = {}
    with engine.begin() as connection:
        connection.exec_driver_sql("PRAGMA foreign_keys=OFF")
        for table in reversed(Base.metadata.sorted_tables):
            connection.execute(table.delete())
        for table in Base.metadata.sorted_tables:
            rows = data.get(table.name) or []
            if not rows:
                imported_counts[table.name] = 0
                continue
            column_names = {column.name for column in table.columns}
            cleaned = [
                {key: value for key, value in row.items() if key in column_names}
                for row in rows
            ]
            connection.execute(table.insert(), cleaned)
            imported_counts[table.name] = len(cleaned)
        connection.exec_driver_sql("PRAGMA foreign_keys=ON")

    attachments_root = Path(ATTACHMENTS_DIR)
    attachments_root.mkdir(parents=True, exist_ok=True)
    restored_attachments = 0
    for name in archive.namelist():
        if not name.startswith(ATTACHMENTS_PREFIX) or name.endswith("/"):
            continue
        relative = name[len(ATTACHMENTS_PREFIX):]
        target = (attachments_root / relative).resolve()
        if attachments_root.resolve() not in target.parents and target != attachments_root.resolve():
            continue  # zip 路径穿越防护
        target.parent.mkdir(parents=True, exist_ok=True)
        target.write_bytes(archive.read(name))
        restored_attachments += 1

    return {
        "table_counts": imported_counts,
        "restored_attachments": restored_attachments,
    }
```

实现注意（执行者必读）：

- 日期/时间列从 JSON 读回时是 ISO 字符串，SQLite 驱动按 TEXT 存也能被 SQLAlchemy `Date`/`DateTime` 读出的前提是格式一致；为稳妥，执行者应在 `import_full_archive` 里对每张表把日期列（`table.columns` 中类型为 `Date`/`DateTime` 的列）用 `datetime.fromisoformat`/`date.fromisoformat` 转回 Python 对象后再 insert——把这个转换写成小工具函数 `_coerce_row_types(table, row)`，勿省略。
- `Base`、`engine` 从 `memory_anki.infrastructure.db.models` 导入（`external_ai_call_logs.py` 第 15 行同款用法）；若 `models.py` 未直接导出 `Base`，改从 `memory_anki.infrastructure.db._tables._base import Base`。

@router.get("/export/full")
def api_export_full(s: Session = Depends(session_dep)):
    zip_bytes, filename = build_full_archive(s)
    return Response(
        zip_bytes,
        media_type="application/zip",
        headers={"Content-Disposition": f"attachment; filename={filename}"},
    )


@router.post("/import/full/preview")
async def api_import_full_preview(file: UploadFile = File(...), s: Session = Depends(session_dep)):
    try:
        return {"ok": True, **inspect_archive(await file.read(), s)}
    except FullTransferError as exc:
        return {"ok": False, "error": str(exc)}


@router.post("/import/full")
async def api_import_full(file: UploadFile = File(...), s: Session = Depends(session_dep)):
    zip_bytes = await file.read()
    try:
        inspect_archive(zip_bytes, s)  # 先校验再做逃生快照，避免坏包也触发备份
        create_rescue_snapshot("before-full-import")
        result = import_full_archive(zip_bytes, s)
        return {"ok": True, **result}
    except FullTransferError as exc:
        return {"ok": False, "error": str(exc)}

import { API_BASE, uploadWithFormData } from '@/shared/api/http'

export const FULL_EXPORT_URL = `${API_BASE}/export/full`

export interface FullImportPreviewResponse {
  ok: boolean
  error?: string
  manifest?: {
    format_version: number
    alembic_revision: string
    created_at: string
    table_counts: Record<string, number>
  }

  schema_match?: boolean
}

export interface FullImportResponse {
  ok: boolean
  error?: string
  table_counts?: Record<string, number>
  restored_attachments?: number
}

export function previewFullImportApi(file: File) {
  const form = new FormData()
  form.append('file', file)
  return uploadWithFormData<FullImportPreviewResponse>('/import/full/preview', form, {
    resourceKey: 'backup:full-import-preview',
    description: '校验全量导入包',
  })
}

export function runFullImportApi(file: File) {
  const form = new FormData()
  form.append('file', file)
  return uploadWithFormData<FullImportResponse>('/import/full', form, {
    resourceKey: 'backup:full-import',
    description: '执行全量导入',
  })
}
